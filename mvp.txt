# services/greenops-engine/src/llm_advisor.py
import httpx
async def generate_recommendation(code: str, energy_kw: float) -> str:
    r = await httpx.post('http://ollama:11434/api/generate', json={
        'model': 'llama3.1:8b',
        'prompt': f'GreenOps: function uses {energy_kw}kW. Optimize:\n{code}',
        'stream': False
    })
    return r.json()['response']
